
p3: Compute the sum of n numbers

function integer computeSum(vector v[integer],integer n)[
	integer sum=0;;
	for { integer i=0;;i<=n;i++}[
		
		sum=sum+v[i];;
	]
]

function integer main()[
	vector v[integer];;
	integer n,sum;;
	read n;;
	for { integer i=0;;i<=n;i++}[
		read v[i];;
	]
	sum=computeSum(v ,n);;
	write sum;;
]
